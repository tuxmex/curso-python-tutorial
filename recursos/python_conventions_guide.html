def calcular_oee(disponibilidad, rendimiento, calidad):
    """
    Calcula el OEE (Overall Equipment Effectiveness).
    
    El OEE es una m√©trica clave en manufactura que mide
    la eficiencia general de un equipo.
    
    Args:
        disponibilidad (float): Porcentaje de tiempo disponible (0-100)
        rendimiento (float): Porcentaje de rendimiento (0-100)
        calidad (float): Porcentaje de calidad (0-100)
    
    Returns:
        float: Valor OEE como porcentaje (0-100)
    
    Example:
        >>> calcular_oee(85.0, 90.0, 95.0)
        72.675
    """
    return (disponibilidad * rendimiento * calidad) / 10000

from enum import Enum
from datetime import datetime
from typing import Dict, List

class NivelAlarma(Enum):
    """Niveles de criticidad seg√∫n ISA-18.2"""
    INFO = "INFORMACION"
    ADVERTENCIA = "ADVERTENCIA"
    ALARMA = "ALARMA"
    CRITICA = "CRITICA"
    EMERGENCIA = "EMERGENCIA"

class SistemaAlarmas:
    """
    Sistema centralizado de gesti√≥n de alarmas industriales.
    
    Cumple con est√°ndares ISA-18.2 para sistemas de alarmas.
    Proporciona logging, escalaci√≥n autom√°tica y notificaciones.
    """
    
    def __init__(self):
        self.alarmas_activas: Dict[str, dict] = {}
        self.historial_alarmas: List[dict] = []
    
    def generar_alarma(self, 
                      equipo_id: str, 
                      mensaje: str, 
                      nivel: NivelAlarma,
                      valor_actual: float = None,
                      limite: float = None) -> None:
        """
        Genera una nueva alarma en el sistema.
        
        Args:
            equipo_id: Identificador del equipo (ej: "L1-TE-001")
            mensaje: Descripci√≥n de la alarma
            nivel: Nivel de criticidad seg√∫n NivelAlarma
            valor_actual: Valor que dispar√≥ la alarma
            limite: L√≠mite que se sobrepas√≥
        """
        timestamp = datetime.now()
        
        alarma = {
            "id": f"{equipo_id}_{timestamp.strftime('%Y%m%d_%H%M%S')}",
            "equipo_id": equipo_id,
            "timestamp": timestamp,
            "mensaje": mensaje,
            "nivel": nivel.value,
            "valor_actual": valor_actual,
            "limite": limite,
            "estado": "ACTIVA",
            "reconocida": False
        }
        
        # Registrar alarma activa
        self.alarmas_activas[alarma["id"]] = alarma
        self.historial_alarmas.append(alarma.copy())
        
        # Escalaci√≥n autom√°tica seg√∫n criticidad
        self._procesar_escalacion(alarma)
        
        # Log de seguridad
        self._log_alarma(alarma)
    
    def _procesar_escalacion(self, alarma: dict) -> None:
        """Procesa la escalaci√≥n seg√∫n el nivel de criticidad."""

            # Parada de emergencia autom√°tica
            self._activar_parada_emergencia(alarma["equipo_id"])

            == NivelAlarma.CRITICA.value:
            # Notificaci√≥n inmediata al operador
            self._notificar_operador(alarma)
        elif nivel == NivelAlarma.ALARMA.value:
            # Registro en log de alarmas
            self._registrar_log_alarma(alarma)
        elif nivel == NivelAlarma.ADVERTENCIA.value:
            # Notificaci√≥n al supervisor
            self._notificar_supervisor(alarma)
        elif nivel == NivelAlarma.INFO.value:
            # Registro informativo
