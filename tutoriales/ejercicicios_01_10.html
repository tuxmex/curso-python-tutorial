"""

Descripci√≥n: Explorar operadores de comparaci√≥n, l√≥gicos y aritm√©ticos

# OPERADORES ARITM√âTICOS
# ==========================================

print(f"{numero} ** 2 = {potencia} (potencia/exponenciaci√≥n)")

# ==========================================
# OPERADORES DE COMPARACI√ìN
# ==========================================

print("\nüîç OPERADORES DE COMPARACI√ìN:")

# Comparaciones que devuelven True o False

es_diferente_0 = numero != 0
es_mayor_30 = numero > 30
es_menor_100 = numero < 100

print(f"¬ø{numero} < 100? {es_menor_100}")
print(f"¬ø{numero} >= 42? {es_mayor_igual_42}")
print(f"¬ø{numero} <= 50? {es_menor_igual_50}")

# ==========================================

# DETECTOR DE CARACTER√çSTICAS ESPECIALES
# ==========================================

print("\nüåü CARACTER√çSTICAS ESPECIALES DEL N√öMERO:")

# M√∫ltiplo de 3

print(f"üî∏ ¬øEs m√∫ltiplo de 3? {es_multiplo_3}")

print(f"üî∏ ¬øEs m√∫ltiplo de 5? {es_multiplo_5}")

# ==========================================

print("\nüé≤ CLASIFICACI√ìN ESPECIAL:")

# N√∫mero en rangos espec√≠ficos

# ==========================================

# ==========================================

        if letra.isalpha():

    # Determinar descripci√≥n
    if puntuacion_final >= 80:
        descripcion = "¬°Nombre de leyenda!"

    else:

def calcular_orbita(altitud_km, radio_tierra=6371):
    """
    Calcula par√°metros orbitales b√°sicos.
    
    Args:
        altitud_km (float): Altitud sobre la superficie en km
        radio_tierra (float): Radio de la Tierra en km
    
    Returns:
        dict: Par√°metros orbitales calculados
    """
    radio_orbital = radio_tierra + altitud_km
    
    # Velocidad orbital (aproximada)
    G = 6.67430e-11  # Constante gravitacional
    M = 5.972e24     # Masa de la Tierra en kg
    velocidad_orbital = sqrt((G * M) / (radio_orbital * 1000)) / 1000  # km/s
    
    # Per√≠odo orbital
    circunferencia = 2 * pi * radio_orbital
    periodo_horas = circunferencia / (velocidad_orbital * 3600)
    
    # Gravedad artificial (aceleraci√≥n centr√≠peta)
    gravedad_artificial = (velocidad_orbital ** 2) / radio_orbital  # km/s¬≤
    gravedad_g = gravedad_artificial / 9.81 * 1000  # en t√©rminos de g terrestre
    
    return {
        "radio_orbital": radio_orbital,
        "velocidad_orbital": velocidad_orbital,
        "periodo_horas": periodo_horas,
        "gravedad_g": gravedad_g
    }

def calcular_distancia_tierra(latitud, longitud):
    """
    Calcula la distancia desde un punto en la Tierra.
    
    Args:
        latitud (float): Latitud en grados
        longitud (float): Longitud en grados
    
    Returns:
        float: Distancia en km
    """
    # Coordenadas de ejemplo de centro de control (Houston)
    lat_houston = 29.7604
    lon_houston = -95.3698
    
    # Convertir a radianes
    lat1, lon1 = radians(lat_houston), radians(lon_houston)
    lat2, lon2 = radians(latitud), radians(longitud)
    
    # F√≥rmula de haversine (simplificada)
    dlat = lat2 - lat1
    dlon = lon2 - lon1
    
    a = sin(dlat/2)**2 + cos(lat1) * cos(lat2) * sin(dlon/2)**2
    c = 2 * math.asin(sqrt(a))
    
    # Radio de la Tierra
    R = 6371  # km
    distancia = R * c
    
    return distancia

def calcular_consumo_combustible(masa_nave, delta_v_requerido):
    """
    Calcula consumo de combustible usando la ecuaci√≥n de Tsiolkovsky.
    
    Args:
        masa_nave (float): Masa de la nave en kg
        delta_v_requerido (float): Cambio de velocidad requerido en m/s
    
    Returns:
        dict: Informaci√≥n sobre combustible
    """
    # Impulso espec√≠fico t√≠pico de motores de hidrazina
    isp = 300  # segundos
    g0 = 9.81  # m/s¬≤
    
    # Ecuaci√≥n de Tsiolkovsky: Œîv = Isp * g0 * ln(m0/m1)
    # Resolviendo para la raz√≥n de masa
    razon_masa = math.exp(delta_v_requerido / (isp * g0))
    masa_combustible = masa_nave * (razon_masa - 1) / razon_masa
    
    return {
        "masa_combustible_kg": masa_combustible,
        "razon_masa": razon_masa,
        "eficiencia": (1 - masa_combustible/masa_nave) * 100
    }

class SensorIndustrial:
    """Sensor industrial con manejo robusto de errores."""
    
    def __init__(self, sensor_id, tipo, rango_min, rango_max):
        self.sensor_id = sensor_id
        self.tipo = tipo
        self.rango_min = rango_min
        self.rango_max = rango_max
        self.estado = "Inicializando"
        self.lecturas = []
        self.errores_consecutivos = 0
        self.max_errores = 3
    
    def leer_sensor(self):
        """
        Simula lectura de sensor con posibles errores.
        
        Returns:
            float or None: Lectura del sensor o None si hay error
        """
        try:
            # Simular fallas ocasionales del sensor
            if random.random() < 0.15:  # 15% probabilidad de falla
                raise ConnectionError(f"Sensor {self.sensor_id} no responde")
            
            # Simular lectura normal
            lectura = random.uniform(self.rango_min - 5, self.rango_max + 5)
            
            # Validar rango
            if lectura < self.rango_min or lectura > self.rango_max:
                raise ValueError(f"Lectura fuera de rango: {lectura:.1f}")
            
            # Lectura exitosa
            self.lecturas.append(lectura)
            self.errores_consecutivos = 0
            self.estado = "Operativo"
            
            print(f"‚úÖ {self.sensor_id} ({self.tipo}): {lectura:.1f}")
            return lectura
            
        except ConnectionError as e:
            self.errores_consecutivos += 1
            print(f"üîå Error de conexi√≥n en {self.sensor_id}: {e}")
            
            if self.errores_consecutivos >= self.max_errores:
                self.estado = "Fuera de servicio"
                print(f"üö® {self.sensor_id} marcado como FUERA DE SERVICIO")
            
            return None
            
        except ValueError as e:
            self.errores_consecutivos += 1
            print(f"üìä Error de rango en {self.sensor_id}: {e}")
            print(f"   Rango v√°lido: {self.rango_min} - {self.rango_max}")
            
            # Usar √∫ltimo valor v√°lido si existe
            if self.lecturas:
                ultimo_valor = self.lecturas[-1]
                print(f"   Usando √∫ltimo valor v√°lido: {ultimo_valor:.1f}")
                return ultimo_valor
            
            return None
            
        except Exception as e:
            self.errores_consecutivos += 1
            print(f"‚ö†Ô∏è Error inesperado en {self.sensor_id}: {e}")
            self.estado = "Error desconocido"
            return None
    
    def obtener_estadisticas(self):
        """Obtiene estad√≠sticas del sensor con manejo de errores."""
        try:
            if not self.lecturas:
                return {"error": "No hay lecturas disponibles"}
            
            promedio = sum(self.lecturas) / len(self.lecturas)
            maximo = max(self.lecturas)
            minimo = min(self.lecturas)
            
            return {
                "sensor_id": self.sensor_id,
                "estado": self.estado,
                "total_lecturas": len(self.lecturas),
                "promedio": round(promedio, 2),
                "maximo": maximo,
                "minimo": minimo,
                "errores_consecutivos": self.errores_consecutivos
            }
            
        except Exception as e:
            return {"error": f"Error calculando estad√≠sticas: {e}"}

def solicitar_numero(mensaje, tipo=float, rango_min=None, rango_max=None, intentos_max=3):
    """
    Solicita un n√∫mero con validaci√≥n robusta.
    
    Args:
        mensaje (str): Mensaje para mostrar al usuario
        tipo (type): Tipo de n√∫mero (int o float)
        rango_min (float): Valor m√≠nimo permitido
        rango_max (float): Valor m√°ximo permitido
        intentos_max (int): M√°ximo n√∫mero de intentos
    
    Returns:
        number or None: N√∫mero v√°lido o None si se agotaron intentos
    """
    for intento in range(intentos_max):
        try:
            # Simular entrada del usuario (en aplicaci√≥n real ser√≠a input())
            if intento == 0:
                entrada_simulada = "abc"  # Primer intento: texto inv√°lido
            elif intento == 1:
                entrada_simulada = "-50"   # Segundo intento: fuera de rango

            print(f"Intento {intento + 1}: '{entrada_simulada}'")

            numero = tipo(entrada_simulada)
            
            # Validar rango
            if rango_min is not None and numero < rango_min:
                raise ValueError(f"Valor debe ser mayor o igual a {rango_min}")
            
            if rango_max is not None and numero > rango_max:
                raise ValueError(f"Valor debe ser menor o igual a {rango_max}")
            
            print(f"‚úÖ Valor v√°lido ingresado: {numero}")
            return numero
            
        except ValueError as e:
            if "could not convert" in str(e) or "invalid literal" in str(e):

            if intento < intentos_max - 1:
                print(f"Por favor, intenta de nuevo ({intentos_max - intento - 1} intentos restantes)")
        
        except Exception as e:
            print(f"‚ùå Error inesperado: {e}")
    
    print(f"‚ùå Se agotaron los intentos. Usando valor por defecto.")
    return None

def calcular_eficiencia_segura(produccion_actual, produccion_objetivo, horas_trabajadas):
    """
    Calcula eficiencia con manejo completo de errores.
    
    Args:
        produccion_actual (int): Piezas producidas
        produccion_objetivo (int): Piezas objetivo
        horas_trabajadas (float): Horas trabajadas
    
    Returns:
        dict: Resultados del c√°lculo o errores
    """
    resultados = {
        "exito": False,
        "errores": [],
        "advertencias": []
    }
    
    try:
        # Validar tipos de datos
        if not isinstance(produccion_actual, (int, float)):
            raise TypeError("Producci√≥n actual debe ser un n√∫mero")
        
        if not isinstance(produccion_objetivo, (int, float)):
            raise TypeError("Producci√≥n objetivo debe ser un n√∫mero")
        
        if not isinstance(horas_trabajadas, (int, float)):
            raise TypeError("Horas trabajadas debe ser un n√∫mero")
        
        # Validar valores l√≥gicos
        if produccion_actual < 0:
            raise ValueError("Producci√≥n actual no puede ser negativa")
        
        if produccion_objetivo <= 0:
            raise ValueError("Producci√≥n objetivo debe ser mayor que cero")
        
        if horas_trabajadas <= 0:
            raise ValueError("Horas trabajadas debe ser mayor que cero")
        
        # Advertencias para valores at√≠picos
        if horas_trabajadas > 24:
            resultados["advertencias"].append("Horas trabajadas excede 24 horas")
        
        if produccion_actual > produccion_objetivo * 1.5:
            resultados["advertencias"].append("Producci√≥n actual muy por encima del objetivo")
        
        # C√°lculos principales
        eficiencia_porcentual = (produccion_actual / produccion_objetivo) * 100
        productividad_hora = produccion_actual / horas_trabajadas
        
        # C√°lculos adicionales con manejo de errores
        try:
            tiempo_por_pieza = horas_trabajadas / produccion_actual if produccion_actual > 0 else 0
        except ZeroDivisionError:
            tiempo_por_pieza = 0
            resultados["advertencias"].append("No se puede calcular tiempo por pieza (producci√≥n = 0)")
        
        # Clasificar rendimiento
        if eficiencia_porcentual >= 100:
            clasificacion = "Excelente"
        elif eficiencia_porcentual >= 90:
            clasificacion = "Muy bueno"
        elif eficiencia_porcentual >= 80:
            clasificacion = "Bueno"
        elif eficiencia_porcentual >= 70:
            clasificacion = "Regular"
        else:
            clasificacion = "Necesita mejora"
        
        # Resultados exitosos
        resultados.update({
            "exito": True,
            "eficiencia_porcentual": round(eficiencia_porcentual, 2),
            "productividad_hora": round(productividad_hora, 2),
            "tiempo_por_pieza": round(tiempo_por_pieza, 3),
            "clasificacion": clasificacion,
            "datos_entrada": {
                "produccion_actual": produccion_actual,
                "produccion_objetivo": produccion_objetivo,
                "horas_trabajadas": horas_trabajadas
            }
        })
        
    except TypeError as e:
        resultados["errores"].append(f"Error de tipo: {e}")
    
    except ValueError as e:
        resultados["errores"].append(f"Error de valor: {e}")
    
    except ZeroDivisionError as e:
        resultados["errores"].append(f"Error de divisi√≥n: No se puede dividir entre cero")
    
    except Exception as e:
        resultados["errores"].append(f"Error inesperado: {e}")
    
    finally:
        # Este bloque SIEMPRE se ejecuta
        timestamp = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        resultados["timestamp"] = timestamp
        print(f"üìù C√°lculo procesado a las {timestamp}")
    
    return resultados

class CelulaTrabajoInteligente:
    """
    üè≠ C√©lula de trabajo que coordina m√∫ltiples robots
    Demuestra composici√≥n: una c√©lula contiene m√∫ltiples robots
    """
    
    def __init__(self, celula_id: str, linea_produccion: str):
        self.celula_id = celula_id
        self.linea_produccion = linea_produccion
        self.robots: List[RobotIndustrial] = []
        self.productos_procesados = 0
        
        print(f"üè≠ C√©lula de trabajo {celula_id} creada en l√≠nea {linea_produccion}")
    
    def agregar_robot(self, robot: RobotIndustrial):
        """ü§ñ Agregar robot a la c√©lula"""
        self.robots.append(robot)
        print(f"‚ûï Robot {robot.robot_id} agregado a c√©lula {self.celula_id}")
    
    def inicializar_celula(self):
        """‚ö° Inicializar todos los robots de la c√©lula"""
        print(f"\n‚ö° Inicializando c√©lula {self.celula_id}...")
        
        for robot in self.robots:
            if robot.encender():
                print(f"   ‚úÖ {robot.robot_id} operativo")
            else:
                print(f"   ‚ùå {robot.robot_id} fall√≥ al inicializar")
        
        print(f"üéØ C√©lula {self.celula_id} lista para producci√≥n")
    
    def procesar_producto(self, especificaciones: Dict) -> bool:
        """üè≠ Procesar un producto completo usando la c√©lula"""
        print(f"\nüè≠ Procesando producto en c√©lula {self.celula_id}")
        print(f"üìã Especificaciones: {especificaciones}")
        
        # Secuencia de producci√≥n usando diferentes robots
        secuencia_exito = True
        
        # Paso 1: Transporte de materiales
        robot_transporte = next((r for r in self.robots if isinstance(r, RobotTransportador)), None)
        if robot_transporte:
            print("\nüì¶ FASE 1: Transporte de materiales")
            if not robot_transporte.ejecutar_tarea(TipoTarea.TRANSPORTE, {
                'origen': [0, 0],
                'destino': [5, 5],
                'peso': especificaciones.get('peso_materiales', 20)
            }):
                secuencia_exito = False
        
        # Paso 2: Ensamble
        robot_ensamble = next((r for r in self.robots if isinstance(r, RobotEnsamblador)), None)
        if robot_ensamble and secuencia_exito:
            print("\nüîß FASE 2: Ensamble de componentes")
            if not robot_ensamble.ejecutar_tarea(TipoTarea.ENSAMBLE, {
                'componentes': especificaciones.get('componentes', ['base', 'tapa']),
                'tornillos': especificaciones.get('tornillos', 4)
            }):
                secuencia_exito = False
        
        # Paso 3: Soldadura (si es requerida)
        robot_soldadura = next((r for r in self.robots if isinstance(r, RobotSoldador)), None)
        if robot_soldadura and especificaciones.get('requiere_soldadura', False) and secuencia_exito:
            print("\nüî• FASE 3: Soldadura")
            if not robot_soldadura.ejecutar_tarea(TipoTarea.SOLDADURA, {
                'longitud_cm': especificaciones.get('longitud_soldadura', 10),
                'espesor_mm': especificaciones.get('espesor_soldadura', 3)
            }):
                secuencia_exito = False
        
        # Paso 4: Inspecci√≥n final
        if secuencia_exito:
            print("\nüîç FASE 4: Inspecci√≥n final")
            robot_inspector = next((r for r in self.robots 
                                  if TipoTarea.INSPECCION in r.obtener_capacidades()), None)
            if robot_inspector:
                if robot_inspector.ejecutar_tarea(TipoTarea.INSPECCION, especificaciones):
                    print("‚úÖ Producto aprobado")
                else:
                    secuencia_exito = False
                    print("‚ùå Producto rechazado en inspecci√≥n final")
        
        # Paso 5: Empaque
        if secuencia_exito and robot_ensamble:
            print("\nüì¶ FASE 5: Empaque")
            robot_ensamble.ejecutar_tarea(TipoTarea.EMPAQUE, {
                'producto': especificaciones.get('nombre_producto', 'Producto Industrial')
            })
        
        if secuencia_exito:
            self.productos_procesados += 1
            print(f"\nüéâ ¬°Producto completado exitosamente en c√©lula {self.celula_id}!")
        else:
            print(f"\n‚ùå Producto fall√≥ en c√©lula {self.celula_id}")
        
        return secuencia_exito
