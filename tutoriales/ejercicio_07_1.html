import datetime
import random
import math
import time
class CentroComandoEspacial:
    def __init__(self):
        self.nombre = "Centro Houston"
        self.tiempo_inicio = datetime.datetime.now()
        self.misiones_activas = []
        self.astronautas = ["Neil A.", "Buzz A.", "Sally R.", "Yuri G."]
        self.planetas = {
            "Marte": {"distancia_km": 225000000, "gravedad": 3.71},
            "Luna": {"distancia_km": 384400, "gravedad": 1.62},
            "Venus": {"distancia_km": 108000000, "gravedad": 8.87}
        }
        
        print(f"üöÄ {self.nombre} inicializado")
        print(f"Hora del sistema: {self.tiempo_inicio.strftime('%Y-%m-%d %H:%M:%S')}")
    
    def tiempo_transcurrido(self):
        ahora = datetime.datetime.now()
        transcurrido = ahora - self.tiempo_inicio
        return transcurrido
    def generar_mision(self):
        tipos_mision = ["Exploraci√≥n", "Investigaci√≥n", "Rescate", "Suministros"]
        tipo = random.choice(tipos_mision)
        destino = random.choice(list(self.planetas.keys()))
        astronauta = random.choice(self.astronautas)
        # Calcular duraci√≥n basada en distancia
        distancia = self.planetas[destino]["distancia_km"]
        velocidad_promedio = 58000  # km/h
        horas_viaje = distancia / velocidad_promedio
        # Agregar variaci√≥n aleatoria
        factor_aleatorio = random.uniform(0.8, 1.2)
        horas_viaje *= factor_aleatorio
        mision = {
            "id": random.randint(1000, 9999),
            "tipo": tipo,
            "destino": destino,
            "astronauta": astronauta,
            "duracion_horas": round(horas_viaje, 2),
            "fecha_lanzamiento": datetime.datetime.now() + datetime.timedelta(days=random.randint(1, 30)),
            "estado": "Preparando",
            "probabilidad_exito": random.uniform(0.85, 0.99)
        }
        self.misiones_activas.append(mision)
        return mision
    def simular_evento_aleatorio(self):
        eventos = [
            ("üåü Descubrimiento cient√≠fico", 0.1),
            ("‚òÑÔ∏è Lluvia de meteoritos detectada", 0.15),
            ("üõ∏ Se√±al extra√±a recibida", 0.05),
            ("‚ö° Tormenta solar", 0.2),
            ("‚úÖ Todo normal", 0.5)
        ]
        
        # Seleccionar evento basado en probabilidades
        rand = random.random()
        acumulado = 0
        
        for evento, probabilidad in eventos:
            acumulado += probabilidad
            if rand <= acumulado:
                return evento
        
        return eventos[-1][0]
    def calcular_trayectoria(self, destino, angulo_lanzamiento):
        distancia = self.planetas[destino]["distancia_km"]
        gravedad_destino = self.planetas[destino]["gravedad"]
        # Convertir √°ngulo a radianes
        angulo_rad = math.radians(angulo_lanzamiento)
        # Velocidad de escape (simplificada)
        velocidad_escape = math.sqrt(2 * 9.81 * 6371000)  # m/s
        # Componentes de velocidad
        vx = velocidad_escape * math.cos(angulo_rad)
        vy = velocidad_escape * math.sin(angulo_rad)
        # Tiempo de vuelo (simplificado)
        tiempo_vuelo = distancia * 1000 / vx  # segundos 
        # Combustible necesario (f√≥rmula simplificada)
        delta_v = gravedad_destino * tiempo_vuelo
        combustible_necesario = delta_v * 0.1  # kg
        return {
            "velocidad_x": vx,
            "velocidad_y": vy,
            "tiempo_vuelo": round(tiempo_vuelo / 3600, 2),  # horas
            "combustible_necesario": round(combustible_necesario, 2)  # kg
        }
    def simular_mision(self, mision):
        print(f"üöÄ Iniciando misi√≥n {mision['id']}: {
            mision['tipo']} a {mision['destino']}")

        print(f"‚è≥ Duraci√≥n estimada: {mision['duracion_hras']} horas")
        print(f"üìÖ Fecha de lanzamiento: {mision['fecha_lanzamiento'].strftime('%Y-%m-%d %H:%M:%S')}")
        print(f"‚úÖ Probabilidad de √©xito: {mision['probabilidad_exito']:.2%}")
        trayectoria = self.calcular_trayectoria(mision['destino'], 45)
        print(f"üõ∞Ô∏è Trayectoria calculada: Velocidad X: {trayectoria['velocidad_x']:.2f} m/s, Velocidad Y: {trayectoria['velocidad_y']:.2f} m/s")
        print(f"‚è±Ô∏è Tiempo de vuelo: {trayectoria['tiempo_vuelo']} horas")
        print(f"üíº Combustible necesario: {trayectoria['combustible_necesario']} kg")
        print("üåå Simulando eventos aleatorios...")
        evento = self.simular_evento_aleatorio()
        print(f"üîÆ Evento aleatorio: {evento}")

        print(f"üéâ Misi√≥n {mision['id']} finalizada")
